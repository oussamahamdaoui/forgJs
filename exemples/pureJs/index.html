Let's try some rules:

Email rule:


  const email = new Rule({
    type: 'email',
    domain: d => ['hotmail', 'outlook', 'gmail'].indexOf(d) !== -1,
  });

Custom rule:


  const PhoneNumber = new Rule({
    type: 'string-int',
    minLength: 10,
    maxLength: 10,
    match: /0(7|6).*/,
  });

Password rule:


  const PassWord = new Rule({
      type: 'password',
      minLength: 8,
      maxLength: 10,
      matchesOneOf: ['@', '_', '-', '.', '?', '$'],
      numbers: 1,
      uppercase: 1,
    });

Url rule:


  const url = new Rule({
    type: 'url',
    domain: domain => domain === 'github.com',
    protocol: protocol => protocol === 'https',
  });

Multiple rules:


  const country = new Rule({
    type: 'string',
    oneOf: ['FR', 'US', 'EN'],
  });
  const zipcode = new Rule({
    type: 'string-int',
    minLength: 5,
    maxLength: 5,
  });

  const street = new Rule({
    type: 'string',
    notEmpty: true,
  });

  const streetNumber = new Rule({
    type: 'string-int',
  });
